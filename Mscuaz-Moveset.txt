local attackDelay = 0.35
local cooldownTime = 0.75
local resetTime = 2
local currentIndex = 1
local isCooldown = false
local lastAttackTime = os.clock()

local toolM1 = Instance.new("Tool")
toolM1.Name = "M1s"
toolM1.RequiresHandle = false
toolM1.Parent = backpack

toolM1.Activated:Connect(function()
    if isCooldown then return end
    isCooldown = true

    events[currentIndex]:FireServer()
    lastAttackTime = os.clock()

    if currentIndex == #events then
        task.wait(cooldownTime)
        currentIndex = 1
    else
        currentIndex += 1
        task.wait(attackDelay)
    end

    isCooldown = false
end)

local function createCooldownTool(name, activateFunction, cooldown)
    local tool = Instance.new("Tool")
    tool.Name = name
    tool.RequiresHandle = false
    tool.Parent = backpack
    local cooldownValue = Instance.new("NumberValue", tool)
    cooldownValue.Value = 0

    local function startCooldown()
        cooldownValue.Value = cooldown
        while cooldownValue.Value > 0 do
            tool.Name = name .. " (" .. string.format("%.1f", cooldownValue.Value) .. "s)"
            task.wait(0.1)
            cooldownValue.Value -= 0.1
        end
        tool.Name = name
    end

    tool.Activated:Connect(function()
        if cooldownValue.Value > 0 then return end
        activateFunction()
        startCooldown()
    end)
end

local function createCooldownTool(name, activateFunction, cooldown)
    local tool = Instance.new("Tool")
    tool.Name = name
    tool.RequiresHandle = false
    tool.Parent = backpack
    local cooldownValue = Instance.new("NumberValue", tool)
    cooldownValue.Value = 0

    local function startCooldown()
        cooldownValue.Value = cooldown
        while cooldownValue.Value > 0 do
            tool.Name = name .. " (" .. string.format("%.1f", cooldownValue.Value) .. "s)"
            task.wait(0.1)
            cooldownValue.Value -= 0.1
        end
        tool.Name = name
    end

    tool.Activated:Connect(function()
        if cooldownValue.Value > 0 then return end
        activateFunction()
        startCooldown()
    end)
end
